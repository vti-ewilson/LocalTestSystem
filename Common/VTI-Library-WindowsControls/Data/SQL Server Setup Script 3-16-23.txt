/**
*	Run as administrative user

*	Change default database location to C:\VTI PC\Databases
*	Change authentication mode to Windows AND SQL Authentication
*	Create vtisa login (password is “V@cuum3342”) and grant it the sysadmin role
*	Disable the default sa login
*	Create vtiuser login (password is “vtiuser”)
*	Create and attach VtiData database
*	Grant only SELECT, UPDATE, INSERT, and DELETE permissions on VtiData database to vtiuser

*	Change App Setting VtiDataConnectionString in Visual Studio Project -> Settings.settings after script done:
*		Data Source=.\SQLEXPRESS;Initial Catalog=VtiData;Persist Security Info=True;User ID=vtiuser;Password=vtiuser
*/

-- Change default user database and user database log directories to C:\VTI PC\Databases
EXEC xp_instance_regwrite N'HKEY_LOCAL_MACHINE'
	, N'SOFTWARE\Microsoft\MSSQLServer\MSSQLServer'
	, N'DefaultData'
	, REG_SZ
	, N'C:\VTI PC\Databases'
GO
EXEC xp_instance_regwrite N'HKEY_LOCAL_MACHINE'
	, N'SOFTWARE\Microsoft\MSSQLServer\MSSQLServer'
	, N'DefaultLog'
	, REG_SZ
	, N'C:\VTI PC\Databases'
GO

--Change authentication mode to Windows AND SQL Authentication
EXEC xp_instance_regwrite 
    N'HKEY_LOCAL_MACHINE', 
    N'Software\Microsoft\MSSQLServer\MSSQLServer', 
    N'LoginMode', 
    REG_DWORD, 
    2;
GO

-- Create vtisa login
CREATE LOGIN vtisa WITH PASSWORD = 'V@cuum3342', CHECK_POLICY = OFF;
GO

-- Grant sysadmin role to vtisa
EXEC master..sp_addsrvrolemember @loginame = N'vtisa', @rolename = N'sysadmin'
GO

-- Disable the default sa login
ALTER LOGIN sa DISABLE ;  
GO  

-- Create vtiuser login
CREATE LOGIN vtiuser WITH PASSWORD = 'vtiuser', CHECK_POLICY = OFF;
GO
