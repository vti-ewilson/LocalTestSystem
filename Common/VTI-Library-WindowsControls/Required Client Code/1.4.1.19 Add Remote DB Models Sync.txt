ManualCommands.cs:

        [ManualCommand("PUSH LOCAL MODELS TO REMOTE DB", true, CommandPermissionType.CheckPermissionWithWarning)]
        public virtual void PushLocalModelsToRemoteDatabase()
        {
            VTIWindowsControlLibrary.VtiLib.PushLocalModelsToRemoteDatabase(Config.Control.SystemTypeForModelsSync.ProcessValue);
        }

        [ManualCommand("PULL MODELS FROM REMOTE DB", true, CommandPermissionType.CheckPermissionWithWarning)]
        public virtual void PullModelsFromRemoteDatabase()
        {
            if (VTIWindowsControlLibrary.VtiLib.PullModelsFromRemoteDatabase(Config.Control.SystemTypeForModelsSync.ProcessValue))
            {
                Config.Control.LastRemoteModelsPullTime.ProcessValue = DateTime.Now.ToString("yyyy/MM/dd h:mm tt");
                Config.Control.Save();
            }
        }
