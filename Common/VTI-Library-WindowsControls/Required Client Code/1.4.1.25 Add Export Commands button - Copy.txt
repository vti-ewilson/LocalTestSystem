1. Double-click on your MainForm.cs file in Solution Explorer. This should display your MainForm in designer mode.
2. At the top of the form, in the menu strip, click on "File". At the bottom of the View menu, there should be an empty box with the text "Type Here".
3. In that box, type in "Export Commands" (without quotes) and press enter to create the new ToolStripMenuItem.
4. Right-click on your new "Export Commands" ToolStripMenuItem and select "Properties".
5. In the properties window, click on the lightning bolt icon (Events button). 
6. Double-click on the empty cell next to the "Click" event to create a click event.
7. Edit your click event to be:

private void exportCommandsToolStripMenuItem_Click(object sender, EventArgs e) {
    Machine.ManualCommands.GenerateBarcodePDF("C:\\VTI PC\\System Software\\barcodes.pdf");
    Machine.ManualCommands.GenerateManual(@"C:\VTI PC\System Software\ManualCommandDescriptions.pdf");
    MessageBox.Show("Manual Command Barcodes saved to C:\\VTI PC\\System Software\\barcodes.pdf\n\nCommand/Parameter Manual saved to C:\\VTI PC\\System Software\\ManualCommandDescriptions.pdf");
}
